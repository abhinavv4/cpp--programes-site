Factorial (Recursive approach):

 
        
#include <iostream>
using namespace std;
int factorial(int n)
{
  if (n <= 1)
    return 1;
  else
    return n * factorial(n - 1);
}
int main()
{
  int num;
  cout << "Enter a positive integer: ";
  cin >> num;
  cout << "Factorial of " << num << " = " << factorial(num) << endl;
  return 0;
}

{
  int n;

  cin >> n;
  int t1 = 0, t2 = 1, nxtterm = 0;
  cout << "Fibonacci series:";
  for (int i = 0; i <= n; ++i)
  {
    cout << t1 << ",";

    t1 = t2;
    t2 = nxtterm;

}

{

  {
    a = x;

operator overloading unary:

 
        
#include <iostream>
using namespace std;
class A 
{
  int a,b;
  public:
  void set(int x, int y)
  {
    a = x;
    b = y;
  }
  void display()
  {
    cout << "a=" << a << " b=" << b;
  }
  A operator++()
  {
    A temp;
    temp.a=++a;
    temp.b=++b;
    return temp;
  }
  A operator++(int)
  {
    A temp;
    temp.a=a++;
    temp.b=b++;
    return temp;
  }            
};
int main()
{
  A obj1,obj2,obj3;
  obj1.set(5,10);
  cout << "Original Data: ";
  obj1.display();
  
  obj2=++obj1;
  cout << "After Pre-increment: ";
  obj2.display();
  
  obj3=obj1++;
  cout << "After Post-increment: ";
  obj3.display();
  
  cout << "Final Data in obj1: ";
  obj1.display();
  return 0;
}

Bineary search in an array:

 
        
//binary search in an array 
#include <iostream>
using namespace std;
int main(){
    int n, key;
    cout << "Enter number of elements: \n";
    cin >> n;
    int arr[n];
    cout << "Enter elements in sorted order: \n";
    for(int i = 0; i < n; i++){
        cin >> arr[i];
    }
    cout << "Enter element to search: \n";
    cin >> key;
    int left = 0, right = n - 1, mid;
    bool found = false;
    while(left <= right){
        mid = left + (right - left) / 2;
        if(arr[mid] == key){
            found = true;
            break;
        }
        else if(arr[mid] < key){
            left = mid + 1;
        }
        else{
            right = mid - 1;
        }
    }
    if(found){
        cout << "Element found at index: " << mid << endl;
    }
    else{
        cout << "Element not found" << endl;
    }
    return 0;
}

Bubble sorting: 

        
#include <iostream>
using namespace std;
int main() {
  int n;
  cout << "Enter number of elements: ";
  cin >> n;
  int arr[n];
  cout << "Enter elements: ";
  for(int i=0; i<n; i++) {
    cin >> arr[i];
  }
  for(int i=0; i<n-1; i++) {
    for(int j=0; j<n-i-1; j++) {
      if(arr[j] > arr[j+1]) {
        int temp = arr[j];
        arr[j] = arr[j+1];
        arr[j+1] = temp;
      }
    }
  }
  cout << "Sorted Array: ";
  for(int i=0; i<n; i++) {
    cout << arr[i] << " ";
  }
  cout << endl;
  return 0;
}

Quick sorting:

 
        
#include <iostream>
using namespace std;
int partition(int arr[], int low, int high)
{
  int pivot = arr[high];
  int i = low - 1;
  for (int j = low; j < high; j++)
  {
    if (arr[j] <= pivot)
    {
      i++;
      swap(arr[i], arr[j]);
    }
  }
  swap(arr[i + 1], arr[high]);
  return i + 1;
}
void quickSort(int arr[], int low, int high)
{
  if (low < high)
  {
    int pi = partition(arr, low, high);
    quickSort(arr, low, pi - 1);
    quickSort(arr, pi + 1, high);
  }
}
int main()
{
  int n;
  cout << "Enter number of elements: ";
  cin >> n;
  int arr[n];
  cout << "Enter elements: ";
  for(int i=0; i<n; i++) {
    cin >> arr[i];
  }
  quickSort(arr, 0, n - 1);
  cout << "Sorted Array: ";
  for(int i=0; i<n; i++) {
    cout << arr[i] << " ";
  }
  cout << endl;
  return 0;
}

Merge sorting:

 
        
#include <iostream>
using namespace std;
void merge(int arr[], int l, int m, int r) {
  int n1 = m - l + 1;
  int n2 = r - m;
  int L[n1], R[n2];
  for (int i = 0; i < n1; i++)
    L[i] = arr[l + i];
  for (int j = 0; j < n2; j++)
    R[j] = arr[m + 1 + j];
  int i = 0, j = 0, k = l;
  while (i < n1 && j < n2) {
    if (L[i] <= R[j]) {
      arr[k] = L[i];
      i++;
    } else {
      arr[k] = R[j];
      j++;
    }
    k++;
  }
  while (i < n1) {
    arr[k] = L[i];
    i++;
    k++;
  }
  while (j < n2) {
    arr[k] = R[j];
    j++;
    k++;
  }
}
void mergeSort(int arr[], int l, int r) {
  if (l < r) {
    int m = l + (r - l) / 2;
    mergeSort(arr, l, m);
    mergeSort(arr, m + 1, r);
    merge(arr, l, m, r);
  }
}
int main() {
  int n;
  cout << "Enter number of elements: ";
  cin >> n;
  int arr[n];
  cout << "Enter elements: ";
  for(int i=0; i<n; i++) {
    cin >> arr[i];
  }
  mergeSort(arr, 0, n - 1);
  cout << "Sorted Array: ";
  for(int i=0; i<n; i++) {
    cout << arr[i] << " ";
  }
  cout << endl;
  return 0;
}

Friend Function:

 
        
#include < iostream & gt;
using namespace std;
class A
{
  int a,b;
  friend void function(A);//friend function declaration
  public:
  void set(int x,int y)
  {
    a=x;
    b=y;
  }
};
void function(A obj)//friend function definition
{
  cout<<"a="<<obj.a<<endl;
  cout<<"b="<<obj.b<<endl;
  cout<<"sum="<<obj.a+obj.b<<endl;
}
int main()
{
  A obj;
  obj.set(10,20);
  function(obj);//friend function call
  return 0;
}
