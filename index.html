Factorial (Recursive approach):

 
        
          #include <iostream>
          using namespace std;
          int factorial(int n)
          {
            if (n <= 1)
              return 1;
            else
              return n * factorial(n - 1);
          }
          int main()
          {
            int num;
            cout <<"Enter a positive integer: ";
            cin >> num;
            cout <<"Factorial of " <<num <<" = " <<factorial(num) <<endl;
            return 0;
          }

Fibonachi Series: 

        
          #include <iostream>
          using namespace std;
          int main()
          {
            int n;
            cout << "enter the position of fibonachi series";
            cin >> n;
            int t1 = 0, t2 = 1, nxtterm = 0;
            cout << "Fibonachi series:";
            for (int i = 0; i <= n; ++i)
            {
              cout<< t1<<",";
              //swapping logic
              nxtterm = t1 + t2;
              t1 = t2;
              t2 = nxtterm;
            }
            cout << endl;
            return 0;
          }

          {
            int a,b;
            public:

            {
              a = x;
              b = y;

            {
              cout <<"a: " <<a <<", b: " <<b <<endl;
            }
            A operator+(A obj)
            {
              A t;
              t.a= a+ obj.a;

              return t;
